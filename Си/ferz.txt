#include <Windows.h>
#include <iostream>
#include <conio.h>
#include <ctime>
using namespace std;

const int size=8;
int field[size][size]={0};
void Draw()
{
	for(int i=0; i<size; ++i)
	{
		for(int j=0; j<size; ++j)
		{
			if(field[i][j]==100)
				cout<<char(1);
			else if(i%2==j%2)
				cout<<char(0xDB);
			else
				cout<<' ';
		}
		cout<<endl;
	}
}

bool Set(int row)
{
	for(int k=0; k<size; ++k)
	{
		if(field[row][k]==0)
		{
			field[row][k]=100;
			if(row==size-1)
				return true;
			for(int i=row+1; i<size; ++i)
			{
				for(int j=0; j<size; ++j)
				{
					if(j==k||row+k==i+j||row-k==i-j)
						++field[i][j];
				}
			}
			if(Set(row+1))
				return true;
			for(int i=row+1; i<size; ++i)
			{
				for(int j=0; j<size; ++j)
				{
					if(j==k||row+k==i+j||row-k==i-j)
						--field[i][j];
				}
			}
			field[row][k]=0;
		}
	}
	return false;
}

void main()
{
	Set(0);
	Draw();
}
