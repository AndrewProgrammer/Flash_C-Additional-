#include <Windows.h>
#include <ctime>
#include <iostream>
#include <conio.h>
using namespace std;
void main()
{
	system("color 1e");
	system("cls");
	HANDLE hOut=GetStdHandle(STD_OUTPUT_HANDLE);
	int c;
	/*do{
		c=_getch();//возвращает код нажатой клавиши
		cout<<c<<' ';
	}while(c!=27);//Esc
	*/
	int j=0, i=0, d;
	int x=1,y=1, sx,sy;
	srand((int)time(0));
	sx=rand()%9+1;
	sy=rand()%9+1;
	do{
		system("cls");
		i=0;
		do{
			j=0;
			do{
				//if(i==0||i==10||j==0||j==10||i==j||i+j==10||i>j&&i+j>10||i<j&&i+j<10)
				//if(i%2==0||j%2==0)
				//if(i%2==j%2)
				if(i==sy&&j==sx)
				{
					SetConsoleTextAttribute(hOut,FOREGROUND_RED|FOREGROUND_INTENSITY|BACKGROUND_BLUE);
					cout<<char(3);
					SetConsoleTextAttribute(hOut,0x1e);
				
				}else if(i==y&&j==x)
					cout<<char(1);
				else if(i==0||i==10||j==0||j==10)
				{
					SetConsoleTextAttribute(hOut,FOREGROUND_GREEN|FOREGROUND_INTENSITY|BACKGROUND_BLUE);
					cout<<char(0xdb);
					SetConsoleTextAttribute(hOut,0x1e);
				}else 
					cout<<' ';
			}while(++j<11);
			cout<<'\n';
		}while(++i<11);
		c=_getch();
		if(c==224||c==0)
			c=_getch();
		switch(c)
		{
		case 75:// <--
			if(x>1) --x;
			break;
		case 77: // -->
			if(x<9) ++x;
			break;
		case 72: // ^
			if(y>1) --y;
			break;
		case 80: // \/
			if (y<9) ++y;
			break;
		}
		d=rand()%4;
		switch(d)
		{
		case 0: sx==1?++sx:--sx; break;
		case 1: sx==9?--sx:++sx; break;
		case 2: sy==1?++sy:--sy; break;
		case 3: sy==9?--sy:++sy; break;
		}
		if(x==sx&&y==sy)
		{
			sx=rand()%9+1;
			sy=rand()%9+1;
		}
	}while(c!=27);//esc
}
