template<typename T>
void myswap(T &a, T &b)
{
	T temp = a;
	a = b;
	b = temp;
}
template < typename T > 
void Sort(T arr, int size)
{
	for (int i = 0; i < size; i++)
	{
		for (int j = size - 1; j > i; j--)
		{
			if (arr[i] > arr[j])
				myswap(arr[i], arr[j]);
		}
	}
}
template<typename T, typename T2>
auto Find(T arr, T2 element, int size) -> decltype(arr)
{
	for (int i = 0; i < size; i++)
	{
		if (arr[i] == element)
		{
			return (arr + i);
		}
	}
	return EXIT_SUCCESS;
}
