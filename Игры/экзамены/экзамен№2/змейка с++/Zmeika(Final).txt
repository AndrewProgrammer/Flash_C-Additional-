void Level(int n)
{
	FILE *f;
	int c = 0;// Хранение текущего считанного символа
	char *path = new char[MAX_PATH];
	if (n == 0)// Режим на выживание
	{
		strcpy(path, "levels/mylevel.txt");
		f = fopen(path, "a+");
	}
	else// Режим карьеры
	{
		char *numlev = new char[3];
		strcpy(path, "levels/level");
		_itoa(n, numlev, 10); // Функция itoa преобразует целое число value в строку string. К цифрам числа value подбираются ANSI символы типа char и записываются в строку string.
		strcat(path, numlev);// добавляет одну строку к другой
		strcat(path, ".txt");
		f = fopen(path, "a+");
		delete[]numlev;
	}
	SetCur(0, 7);// Отрисовка игрового поля
	setColor(GREEN, BLACK);
	for (int y = 0; y < height + 2; ++y)
	{
		for (int x = 0; x < width + 2; ++x)
		{
			if (x == 0 || y == 0 || x == width + 1 || y == height + 1)
				cout << char(219);
			else
				cout << " ";
		}
		cout << "\n";
	}
	for (int i = 0; i < height; ++i)// Обнуление массива
	{
		for (int j = 0; j < width; ++j)
		{
			field[i][j] = 0;
		}
	}
	bool error = 0;
	for (int i = 0; i < height; ++i)// Считывание с файла
	{
		for (int j = 0; j < width; ++j)
		{
			if (feof(f))// Преждевременный конец файла
			{
				fclose(f);// Закрытие файла
				f = fopen(path, "w+");// Открытие и перезапись файла
				for (int i = 0; i < height; ++i)// Обнуление массива
				{
					for (int j = 0; j < width; ++j)
					{
						field[i][j] = 0;
					}
				}
				for (int i = 0; i < height; ++i)// Заполнение файла нулями
				{
					for (int j = 0; j < width; ++j)
					{
						fputc('0', f);//Функция fputc записывает символ в поток и перемещает позицию индикатора положения.
					}
					if (i == height)
						break;
					fputc('\n', f);
				}
				SetCur(0, 0);
				cout << " Error" << path << "\nPress anykey for continue";
				_getch();
				SetCur(0, 0);
				line(100, ' ');
				SetCur(0, 1);
				line(100, ' ');
				fclose(f);
				error = 1;
				break;
			}
			c = fgetc(f);//Функция fgetc возвращает символ на который ссылается внутренний индикатор позиции файла указанного потока
			if ((int)c != 13 && (int)c != 10)// Фильтр символа переноса строки и перевода каретки
			{
				field[i][j] = c - 48;// Т.к считывается код символа, нужно отнять 48 для получение цифры
			}
			else
				--j;// Возврат счетчика если встретился символ переноса строки и перевода каретки
		}
		if (error)// При ошибке остановить считывание
			break;
	}
	field[27][39] = 0;// Освобождение места впереди змеи
	field[26][39] = 0;
	field[25][39] = 0;
	field[28][39] = 0;
	delete[]path;// Освободить память
	fclose(f);// Закрыть файл

	setColor(GREEN, BLACK);
	for (int y = 8; y < height + 8; ++y)// Отрисовка уровня
	{
		for (int x = 1; x < width + 1; ++x)
		{
			SetCur(x, y);
			if (field[y - 8][x - 1] == 1)
			{
				setColor(WHITE, BLACK);
				cout << '*';
				setColor(GREEN, BLACK);
			}
			else
				cout << " ";
		}
		cout << "\n";
	}
}

	for (int i = 0; i < height; i++)// Считывание с файла
	{
		for (int j = 0; j < width; j++)
		{
			if (feof(f))// Преждевременный конец файла

				fclose(f);// Закрытие файла

				for (int i = 0; i < height; i++)// Обнуление массива

						field[i][j] = 0;
					}

				_getch();

				line(100, ' ');
				SetCur(0, 1);

			s = fgetc(f);
			if (s != 13 && s != 10)// Фильтр символа переноса строки и перевода каретки
			{
				field[i][j] = s - 48;// Т.к считывается код символа, нужно отнять 48 для получение цифры

					setColor(WHITE, BLACK);
					cout << '*';
					setColor(GREEN, BLACK);
				}
				else
					cout << " ";
			}
			else
				--j;// Возврат счетчика если встретился символ переноса строки и перевода каретки
		}
		if (error)
			break;
	}
	HANDLE h2 = GetStdHandle(STD_INPUT_HANDLE);// Дескриптор ввода
	SetConsoleMode(h2, ENABLE_MOUSE_INPUT);// Включить режим использования меню

	setColor(GREEN, BLACK);

					cout << brush;
				}
				else
				{
					SetConsoleCursorPosition(h, c);
					field[c.Y - 8][c.X - 1] = 0;
					cout << ' ';
				}
			}
			else if (all_events[i].Event.MouseEvent.dwButtonState == FROM_LEFT_1ST_BUTTON_PRESSED && c.Y == 14 && c.X >= 86 && c.X <= 94)// Выбор инструмента
			{
				if (brush == '*')
				{

					setColor(WHITE, BLACK);

					brush = '#';
				}
				else
				{
					SetCur(86, 14);
					setColor(WHITE, BLACK);
					cout << "Brush";
					brush = '*';
				}

			else if (all_events[i].Event.MouseEvent.dwButtonState == FROM_LEFT_1ST_BUTTON_PRESSED && c.Y == 18 && c.X >= 86 && c.X <= 94)// Очистить экран

				{

					{
						if (x == 0 || y == 0 || x == width + 1 || y == height + 1)
							cout << char(219);
						else
							cout << " ";
					}
					cout << "\n";
				}
				setColor(WHITE, BLACK);
			}
			else if (all_events[i].Event.MouseEvent.dwButtonState == FROM_LEFT_1ST_BUTTON_PRESSED && c.Y == 22 && c.X >= 86 && c.X <= 94)// Выход

				system("cls");
				return;
			}

			{
				FILE *f;
				char *ps = "\n";
				SetCur(1, 3);

				f = fopen("levels/mylevel.txt", "w+");
				for (int i = 0; i < height; ++i)
				{
					for (int j = 0; j < width; ++j)
					{
						char *s = new char[2];
						_itoa(field[i][j], s, 10);
						fputs(s, f);
						delete s;
					}
					if (i == height - 1)
						break;
					fputc(*ps, f);
				}
				fclose(f);
				system("cls");
				return;
			}
		}
	}

					setColor(BLUE, BLACK);
				else if (SuperSpeedflag)
					setColor(DARKYELLOW, BLACK);
				else if (delete2flag)
					setColor(DARKBLUE, BLACK);
				else
					setColor(DARKGREEN, BLACK);
				cout << char(253);
			}
		}
		if (SuperSpeedflag)
			Sleep(speed2);
		else
			Sleep(speed);
		if (delete2flag)
		if (lengthSnake % 2 == 0)
		{
			lengthSnake / 2;
			pointsTemp = -1;
			livessnake = 1;
		}
		else
			lengthSnake = startLengthSnake;
	}
}
