#include<iostream>
#include <time.h>
using namespace std;

{
	setlocale(0, "ru");
	const int size = 10;
	int b=0;
	double a[size] = { 5.1, 2.6, 2.8, 3, 4, 10, 3.8, 7.9, 9, 1};
	for (int i = 0; i < size; i++)
	{
		cout << a[i] << ' ';

			b++;
	}

}
