#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	int max, min;
	cout << "Создать диапозон от максимального до минимального\nВведите максимальное число диапозона и минимальное\n";
	cin >> max >> min;
	srand(time(0));
	rand();
	const int size = 15;
	int b[size];
	for (int k = 0; k < size; k++)
	{
		b[k] = rand() % (max - min + 1) + min;
	}
	cout << "Массив с рандомными числами :\n";
	for (int j = 0; j < 15; j++)
	{
		cout << b[j] << ' ';
	}
	cout << endl;
}
