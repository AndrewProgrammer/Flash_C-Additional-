#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	rand();
	const int size = 15;
	int b[size];
	for (int k = 0; k < size; k++)
	{
		b[k] = rand() % 101;
	}
	cout << "Массив с рандомными числами:\n";
	for (int j = 0; j < size; j++)
	{
		cout << b[j] << ' ';
	}
	cout << endl;
	for (int i = 0; i < size - 1; i++)
	{
		int min = i;
		for (int j = i + 1; j < size; j++)
		{
			if (b[j] < b[min])
			{
				min = j;
			}
		}
		swap(b[i], b[min]);
	}
	cout << "Отсортированый массив по возрастанию:\n";
	for (int k = 0; k < size; k++)
	{
		cout << b[k] << ' ';
	}
	cout << endl;
}
