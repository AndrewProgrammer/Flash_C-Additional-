#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	srand(time(0));
	rand();
	const int size = 15;
	int b[size];
	for (int k = 0; k < size; k++)
	{
		b[k] = rand() % 101;
	}
	cout << "Массив с рандомными числами:\n";
	for (int j = 0; j < size; j++)
	{
		cout << b[j] << ' ';
	}
	cout << endl;
	for (int i = 1; i < size; i++)
	{
		for (int j = i; j > 0 && b[j] < b[j - 1]; j--)
		{
			swap(b[j], b[j - 1]);
		}
	}
	cout << "Отсортированый массив методом вставками:\n";
	for (int k = 0; k < size; k++)
	{
		cout << b[k] << ' ';
	}
	cout << endl;
	int a;
	cin >> a;
	int start = 0, end = size - 1, pos = -1;
	while (start <= end)
	{
		int middle = (start + end) / 2;
		if (a == b[middle])
		{
			pos = middle;
			break;
		}
		else if (a > b[middle])
		{
			start = middle + 1;
		}
		else 
		{
			end = middle - 1;
		}
	}
	cout << pos << endl;
}
