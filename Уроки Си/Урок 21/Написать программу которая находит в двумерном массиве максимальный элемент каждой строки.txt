#include <iostream>
#include <time.h>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int m = 3;
	const int n = 2;
	int A[m][n];
	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			A[i][j] = rand() % 101;
			cout << A[i][j] << ' ';
		}
		cout << endl;
	}
	cout << endl;
	for (int i = 0; i < m; i++)
	{
		int max = A[i][0];
		for (int j = 0; j < n; j++)
		{
			if (A[i][j] > max)
			{
				max = A[i][j];
			}
		}
		cout << "Максимальный элемент " << i << "-ой строки = " << max << endl;
	}
}
