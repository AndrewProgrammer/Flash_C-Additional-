#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int size = 10;
	int a[size], min = -5, max = 5;
	srand(time(0));
	rand();
	for (int k = 0; k < size; k++)
	{
		a[k] = rand() % (max - min + 1) + min;
	}
	cout << "Массив с рандомными числами: \n";
	for (int j = 0; j < size; j++)
	{
		cout << a[j] << ' ';
	}
	cout << endl;
	for (int i = 0; i < size; i++)
	{
		if (a[i] == 0)
		{
			for (int d = i; d < size - 1; d++)
			{
				a[d] = a[d + 1];
			}
			a[size - 1] = -1;
		}
	}
	cout << "Массив №2 без нулей числами: \n";
	for (int j = 0; j < size; j++)
	{
		cout << a[j] << ' ';
	}
	cout << endl;
}
