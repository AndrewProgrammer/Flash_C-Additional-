#include <time.h>
#include <iostream>
using namespace std;
void main()
{
	setlocale(0, "ru");
	const int size = 10;
	int a[size],min = -20, max = 20;
	srand(time(0));
	rand();
	for (int k = 0; k < size; k++)
	{
		a[k] = rand() % (max - min + 1) + min;
	}
	cout << "Массив с рандомными числами: \n";
	for (int j = 0; j < size; j++)
	{
		cout << a[j] << ' ';
	}
	cout << endl;
	for (int i = 0; i < size; i++)
	{
		if (a[i] < 0)
		{
			a[i] *= -1;
		}
	}
	cout << "Массив №2 с положительными числами: \n";
	for (int j = 0; j < size; j++)
	{
		cout << a[j] << ' ';
	}
	cout << endl;
}
